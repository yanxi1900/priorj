#(US03.txt) User Story 03  - Create project and set paths. 
# warning: before run tests adjust the paths.

resetSystem

# valid input
createProject name="xpto" version="junit4"

setPathApp pathApp="/home/samuel/eclipseEE/workspace/PriorJ/resources/app/"
setPathCode pathCode="/home/samuel/eclipseEE/workspace/PriorJ/resources/app/src"
setPathLib pathLib="/home/samuel/eclipseEE/workspace/PriorJ/resources/app/lib"
setPathTest pathTest="/home/samuel/eclipseEE/workspace/PriorJ/resources/app/test"
setPathNew pathNew="/home/samuel/eclipseEE/workspace/PriorJ/resources/appnew/src"

expect "xpto" getAttributeOpenedProject attribute="name"
expect "JUnit 4.x" getAttributeOpenedProject attribute="version"
expect "" getAttributeOpenedProject attribute="pathApp"
expect "" getAttributeOpenedProject attribute="pathCode"
expect "" getAttributeOpenedProject attribute="pathLib"
expect "" getAttributeOpenedProject attribute="pathTest"
expect "" getAttributeOpenedProject attribute="pathNew"

#invalid conditions

expectError "Empty application path" setPathApp pathApp=""
expectError "Invalid application path Null" setPathApp pathApp=
expectError "Application path not found" setPathApp pathApp="/home/samuel/app2"

expectError "Empty code path" setPathCode pathCode=""
expectError "Invalid code path Null" setPathCode pathCode=
expectError "Code path not found" setPathCode pathCode="/home/samuel/app2/src"

expectError "Invalid library path Null" setPathLib pathLib=
expectError "Library path not found" setPathLib pathLib="/home/samuel/app2/lib"

expectError "Empty test path" setPathTest pathTest=""
expectError "Invalid test path Null" setPathTest pathTest=
expectError "Test path not found" setPathTest pathTest="/home/samuel/app2/test"

closeOpenProject

expect false hasOpenedProject
