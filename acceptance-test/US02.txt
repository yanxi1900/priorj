# (US02.txt) User Story 02  -  Open , Search and close projects. 

resetSystem

# valid input
createProject name="xpto01" version="junit4"
createProject name="xpto02" version="junit3"
createProject name="xpto03" version="junit4"

#invalid conditions
expectError "Invalid input" createProject name= version="junit4"
expectError "Invalid input" createProject name="xpto" version=

expectError "Project name can not be empty" createProject name="" version="junit3"
expectError "Empty version name" createProject name="xpto" version=""
expectError "Invalid junit version name" createProject name="xpto" version="xpto"

# valid conditions search project
expect true searchProject projectName="xpto01"
expect true searchProject projectName="xpto02"
expect true searchProject projectName="xpto03"
expect false searchProject projectName="x.pto.01"

#invalid conditions to search
expectError "Empty Project Name" searchProject projectName=""
expectError "Invalid Project Name" searchProject projectName=

#valid open project
expect true hasOpenedProject

expect "xpto03" getAttributeOpenedProject attribute="name"
expect "JUnit 4.x" getAttributeOpenedProject attribute="version"
expect "" getAttributeOpenedProject attribute="pathApp"
expect "" getAttributeOpenedProject attribute="pathCode"
expect "" getAttributeOpenedProject attribute="pathLib"
expect "" getAttributeOpenedProject attribute="pathTest"
expect "" getAttributeOpenedProject attribute="pathNew"

#invalid conditions 
expectError "Empty Project Attribute" getAttributeOpenedProject attribute=""
expectError "Invalid Attribute Null" getAttributeOpenedProject attribute=
expectError "Nonexistent Project Attribute" getAttributeOpenedProject attribute="anything"

closeOpenProject

expect false hasOpenedProject

expect 3 numberOfProjects

removeProject name"xpto01"
removeProject name"xpto02"
removeProject name"xpto03"

expect 0 numberOfProjects

quit
